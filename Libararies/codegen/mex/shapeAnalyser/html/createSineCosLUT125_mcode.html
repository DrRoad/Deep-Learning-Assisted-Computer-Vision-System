262%CREATE SINE AND COSINE LOOKUP TABLE (IMPROVES SPEED OF COMPUTATION)
263%--------------------------------------------------------------------------
264function [sineTable cosTable] = createSineCosLUT(Tethas)
265tethaRange = length(Tethas);
266sineTable = zeros(1,tethaRange);
267cosTable = zeros(1,tethaRange);
268for count = 1:tethaRange
269    sineTable(1,count) = sind(Tethas(1,count));
270    cosTable(1,count) = cosd(Tethas(1,count));
271end
272end %end of LUT generation function
