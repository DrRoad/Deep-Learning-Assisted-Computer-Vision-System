1function [detectedRegion available ROI] = getROI(img, windowSize, stepSize, beacon)
  2%{
  3--------------------------------------------------------------------------
  4ROIExtractor Used to extract from a scene, the region of interest wherein 
  5%the  traffic sign classification needs to take place.
  6
  7   Created on: 16th June, 2017
  8   Author: Oluwole Oyetoke Jnr
  9   Using MATLAB 2016
 10
 11STEPS
 121. Get image in RGB
 132. Transform from RGB to HSV space
 143. Turn image to binary through colour segmentation (red, blue & yellow)
 154. Use Connected component analysis technique + Union Find 
 16Algorithm to determine Region of Interest (ROI)
 175. If ROI is available, crop image to ROI, return ROI as detected region and
 18also return the co-ordinates of the detected region wrt the original image 
 19---------------------------------------------------------------------------
 20%}
 21 
 22%Input Validation
 23%if nargin ~= 4
 24%    error('getROI:Input_Argument_Error','This function works with 1 input argument -img, windowSize, stepSize, beacon- ')
 25%end

 28
 29%{
 30CONVERT RGB IMAGE TO HSV SPACE
 31---------------------------------------------------------------------------
 32%H - Hue - contains the original colour itself
 33%S - Saturation - contains the information on number of pixels available for a colour
 34%V - Value - contains information related to brightness or darkness of a pixel
 35
 361. Automatic method i.e allowing MATLAB do the magic
 37%hsvImg = rgb2hsv(img); 
 38
 392. Manual method (angular solution)  
 40Iterate through picture.....
 41        numerator  = 0.5 * ((red-green)+(red-blue));
 42        left = (red-green)*(red-green);
 43        right = (red-blue)*(green-blue);
 44        denominator = sqrt(left+right);
 45        hue = acosd(numerator/denominator);
 46        saturation = 1-((3/(red+green+blue))*minimum);
 47.........
 48
 49---------------------------------------------------------------------------
 50%}
 51
 52%3. Manual Methiod 2 - Actual way RGB to HSV happens
 53    %Normalize image to the range of [0 1] by dividing each value by 255 
 54    hsvImg = rgb2hsv(img);
 55    [xDim, yDim, zDim] = size(hsvImg);
 56
 57
 58    %Normalize HSV[0, 1]
 59    normalizedHue = hsvImg(:,:,1);
 60    normalizedSaturation = hsvImg(:,:,2); 
 61    normalizedValue = hsvImg(:,:,3);
 62    
 63    %Threshold
 64    redSegemnt1Mask = (normalizedHue > 0.8) & (normalizedHue <= 1) & (normalizedSaturation>0.48) & (normalizedSaturation <= 1) & (normalizedValue>=0.4) & (normalizedValue<=1);
 65    redSegemnt2Mask = (normalizedHue > 0) & (normalizedHue < 0.05) & (normalizedSaturation>0.48) & (normalizedSaturation <= 1) & (normalizedValue>=0.4) & (normalizedValue<=1);
 66    blueSegemntMask = (normalizedHue >=0.52) & (normalizedHue <= 0.64) & (normalizedSaturation>0.7) & (normalizedSaturation <= 1) & (normalizedValue>=0.1953) & (normalizedValue<=1);
 67    yellowSegemntMask = (normalizedHue >0.10) & (normalizedHue < 0.14) & (normalizedSaturation>0.4) & (normalizedSaturation < 1) & (normalizedValue>0.8) & (normalizedValue<1);
 68    
 69   overalMask = redSegemnt1Mask | redSegemnt2Mask | blueSegemntMask | yellowSegemntMask;
 70   hsvImg(:,:,1) =  hsvImg(:,:,1) & overalMask;
 71   hsvImg(:,:,2) =  hsvImg(:,:,2) & overalMask;
 72   hsvImg(:,:,3) =  hsvImg(:,:,3) & overalMask;
 73   regionShed = overalMask;
 74    
 75 
 76%{
 77---------------------------------------------------------------------------
 78SLIDING WINDOW DETECTION
 791. Step size showl not be greater than window size 
 802. Use the below to get a list of possible step and window sizes combos
 813. Beacon size is used to indicate that if x% of the region or more is bright,
 82 then assume region to be a positive find.
 834. Perform pixel sum pooling per window. If sum>=beacon, then mark region
 84to be a possibility
 85
 86for winSize=50:xDim-1
 87    for stpSz =2:xDim-1
 88    CheckX = mod((xDim-winSize),stpSz);
 89    CheckY = mod((yDim-winSize),stpSz);
 90    if CheckX==0 && CheckY==0
 91       sprintf('Possible Window Sizes (>50) and Corresponding Step Size (Apart from 1) -> (%d,%d)', winSize, stpSz)
 92    end
 93    end
 94end
 95%}
 96
 97stepSizeXCheck = mod((xDim-windowSize),stepSize);
 98stepSizeYCheck = mod((yDim-windowSize),stepSize); 
 99if stepSize>windowSize
100    error('getROI:Step_Size_error','Please select a suitable step size less than the window size');
101elseif stepSizeXCheck~=0
102     error('getROI:Step_Size_error','Selected step size does not fit for the X axis');
103elseif stepSizeYCheck~=0
104     error('getROI:Step_Size_error','Selected step size does not fit for the Y axis');
105end
106
107%Initialize aprameters and variables
108startX = 1; startY = 1; stopX = windowSize; stopY = windowSize;
109xBlocks = ((xDim-windowSize)/stepSize)+1;
110yBlocks = ((yDim-windowSize)/stepSize)+1;
111xCounter=1;
112yCounter=1;
113
114hueCheck=0; saturationCheck=0; valueCheck=0;
115regionSum=0; %Sum of the number of Pixels which are potentialy members of the ROI
116
117totalBlocks = xBlocks * yBlocks;
118regionMap = zeros(xBlocks,yBlocks);
119%regionMap = gpuArray(regionMap1);
120Map = MapRegion;
121Map(xBlocks,yBlocks) = MapRegion;
122regionSizeTracker = zeros(1,totalBlocks); 
123%regionSizeTracker= gpuArray(regionSizeTracker1);
124
125myLeftNeighVal =0; myRightNeighVal = 0; myTopNeighVal = 0; myBottomNeighVal=0;
126myLeftNeigRegion =-1; myRightNeighRegion = -1; myTopNeighRegion = -1; myBottomNeighRegion=-1;
127regionCounter=1;
128regionsum2=0;
129
130
131 for BlockLoop = 1:totalBlocks
132      regionSum=0;  
133      xCount = xCounter;
134      yCount = yCounter;
135      
136      %Pixel Sum Pooling happens here 
137     regionSum = sum(sum(regionShed(startX:stopX, startY:stopY)));
138 
139     %Save region's potential & Link Regions Together
140     regionMap(xCounter, yCounter) = regionSum;
141    
142     Map(xCount, yCount).value = regionSum;
143     Map(xCount,yCount).xCoord = xCount;
144     Map(xCount,yCount).yCoord = yCount;
145      if xCount==1 && yCount ==1
146         Map(xCount,yCount).left=Map(xCount,yCount); 
147         Map(xCount,yCount).right=Map(xCount+1,yCount);
148         Map(xCount,yCount).top=Map(xCount,yCount);
149         Map(xCount,yCount).bottom=Map(xCount,yCount+1);
150      elseif xCount==xBlocks && yCount ==1
151         Map(xCount,yCount).left=Map(xCount-1,yCount); 
152         Map(xCount,yCount).right=Map(xCount,yCount);
153         Map(xCount,yCount).top=Map(xCount,yCount);
154         Map(xCount,yCount).bottom=Map(xCount,yCount+1);
155      elseif xCount==1 && yCount==yBlocks
156         Map(xCount,yCount).left=Map(xCount,yCount); 
157         Map(xCount,yCount).right=Map(xCount+1,yCount);
158         Map(xCount,yCount).top=Map(xCount,yCount-1);
159         Map(xCount,yCount).bottom=Map(xCount,yCount);
160      elseif xCount==xBlocks && yCount==yBlocks
161         Map(xCount,yCount).left=Map(xCount-1,yCount); 
162         Map(xCount,yCount).right=Map(xCount,yCount);
163         Map(xCount,yCount).top=Map(xCount,yCount-1);
164         Map(xCount,yCount).bottom=Map(xCount,yCount);
165      elseif xCount==1
166         Map(xCount,yCount).left=Map(xCount,yCount); 
167         Map(xCount,yCount).right=Map(xCount+1,yCount);
168         Map(xCount,yCount).top=Map(xCount,yCount-1);
169         Map(xCount,yCount).bottom=Map(xCount,yCount+1);
170      elseif xCount==xBlocks
171         Map(xCount,yCount).left=Map(xCount-1,yCount); 
172         Map(xCount,yCount).right=Map(xCount,yCount);
173         Map(xCount,yCount).top=Map(xCount,yCount-1);
174         Map(xCount,yCount).bottom=Map(xCount,yCount+1);
175      elseif yCount==1
176         Map(xCount,yCount).left=Map(xCount-1,yCount); 
177         Map(xCount,yCount).right=Map(xCount+1,yCount);
178         Map(xCount,yCount).top=Map(xCount,yCount);
179         Map(xCount,yCount).bottom=Map(xCount,yCount+1);
180      elseif yCount==yBlocks
181         Map(xCount,yCount).left=Map(xCount-1,yCount); 
182         Map(xCount,yCount).right=Map(xCount+1,yCount);
183         Map(xCount,yCount).top=Map(xCount,yCount-1);
184         Map(xCount,yCount).bottom=Map(xCount,yCount);
185      else
186         Map(xCount,yCount).left=Map(xCount-1,yCount);
187         Map(xCount,yCount).right=Map(xCount+1,yCount);
188         Map(xCount,yCount).top=Map(xCount,yCount-1);
189         Map(xCount,yCount).bottom=Map(xCount,yCount+1);
190      end
191      
192     %Check to see if current region is bright enough
193     if Map(xCount, yCount).value>beacon %i.e more than 1/4th of the region is white
194         %Check the regions its neighbours belong to
195         myLeftNeighRegion =  Map(xCount, yCount).left.region; 
196         myRightNeighRegion =  Map(xCount, yCount).right.region; 
197         myTopNeighRegion =  Map(xCount, yCount).top.region; 
198         myBottomNeighRegion =  Map(xCount, yCount).bottom.region;
199         
200         %If any of its neighbours already belong to a region, select the region of the neigbour with the lowest value and force this one to join it
201         if myLeftNeighRegion>0 || myRightNeighRegion>0 || myTopNeighRegion>0 || myBottomNeighRegion>0 
202         A = [myLeftNeighRegion myRightNeighRegion myTopNeighRegion myRightNeighRegion];
203         A1 = A(find(A~=-1)); %-1 regions are black areas of the picture or regions that have not been attende to yet
204         oldestRegion = min(A1);
205         Map(xCount, yCount).region = oldestRegion;
206         regionSizeTracker(1,oldestRegion) = regionSizeTracker(1,oldestRegion)+1;
207         
208         %Force other neigbours of this region that may be already part of another region to join the lower region
209         if myLeftNeighRegion >0 && myLeftNeighRegion~=oldestRegion
210             regionSizeTracker(1,myLeftNeighRegion) = regionSizeTracker(1,myLeftNeighRegion)-1; %Update region Size tracker
211             Map(xCount, yCount).left.region = oldestRegion;
212             regionSizeTracker(1,oldestRegion) = regionSizeTracker(1,oldestRegion)+1; %Update region Size tracker
213         end
214         if myRightNeighRegion >0 && myRightNeighRegion~=oldestRegion
215             regionSizeTracker(1,myRightNeighRegion) = regionSizeTracker(1,myRightNeighRegion)-1; %Update region Size tracker
216             Map(xCount, yCount).right.region = oldestRegion;
217             regionSizeTracker(1,oldestRegion) = regionSizeTracker(1,oldestRegion)+1;  %Update region Size tracker
218         end
219         if myTopNeighRegion >0 && myTopNeighRegion~=oldestRegion
220            regionSizeTracker(1,myTopNeighRegion) = regionSizeTracker(1,myTopNeighRegion)-1; %Update region Size tracker
221             Map(xCount, yCount).top.region = oldestRegion;
222             regionSizeTracker(1,oldestRegion) = regionSizeTracker(1,oldestRegion)+1;  %Update region Size tracker
223         end
224         if myBottomNeighRegion >0 && myBottomNeighRegion~=oldestRegion
225             regionSizeTracker(1,myBottomNeighRegion) = regionSizeTracker(1,myBottomNeighRegion)-1; %Update region Size tracker
226             Map(xCount, yCount).bottom.region = oldestRegion;
227             regionSizeTracker(1,oldestRegion) = regionSizeTracker(1,oldestRegion)+1;  %Update region Size tracker
228         end
229          
230         %If non of its neigbour already belongs to a region, then assign it a region
231         else
232         Map(xCount, yCount).region = regionCounter;
233         regionSizeTracker(1,regionCounter) = regionSizeTracker(1,regionCounter)+1; 
234         regionCounter = regionCounter+1;
235         
236         end %End of 'if any of its neighbours already....'
237       
238     else
239      %Do nothing
240     end %End of 'check to see if current region is bright enough.....'
241    
242     
243    %Loop management 
244     if yCounter==yBlocks
245        %reset xStart, xStop positions and xCounter 
246         startY = 1;
247         stopY = windowSize;
248         yCounter=0;
249         xCounter = xCounter+1;
250         
251         startX=startX+stepSize;
252         stopX=startX+windowSize-1;
253     else
254       startY = startY+stepSize;
255       stopY = startY+windowSize-1; 
256     end
257
258     
259     yCounter=yCounter+1;
260 end %end of xBlockLoop 
261
262 
263 
264 %Get the region with the highest potential
265 [val index] = max(regionSizeTracker);
266 
267%{
268 currentHighestX=-1; currentLowestX=xDim+1; currentHighestY=-1; currentLowestY=yDim+1; thisX=0; thisY=0;
269 xCnt=1; yCnt=1;
270 
271 mapFound = findobj(Map, 'region', index);
272 if isempty(mapFound)
273  %Do nothing   
274  sprintf('Region Not Found')
275 else
276    len = length(mapFound);
277    for pos = 1:len
278    maxX = max(mapFound(pos).xCoord); 
279    minX = min(mapFound(pos).xCoord); 
280    maxY = max(mapFound(pos).yCoord); 
281    minY = min(mapFound(pos).yCoord); 
282        if maxX>=currentHighestX
283            currentHighestX = maxX;
284        end
285        if minX<=currentLowestX
286            currentLowestX = minX; 
287        end
288        
289        if maxY>=currentHighestY
290            currentHighestY = maxY;
291        end
292       if minY<=currentLowestY
293            currentLowestY = minY; 
294        end
295    end
296 end
297%}
298 
299
300 currentHighestX=-1; currentLowestX=xDim+1; currentHighestY=-1; currentLowestY=yDim+1; thisX=0; thisY=0;
301 xCnt=1; yCnt=1;
302  for length2 = 1:totalBlocks
303  
304     if Map(xCnt,yCnt).region==index
305        thisX = Map(xCnt,yCnt).xCoord; 
306        thisY = Map(xCnt,yCnt).yCoord;
307        
308        if thisX>currentHighestX 
309           currentHighestX = thisX; 
310        elseif thisX<currentLowestX
311            currentLowestX = thisX;
312        else
313        end
314        
315        if thisY>currentHighestY 
316           currentHighestY = thisY; 
317        elseif thisY<currentLowestY
318            currentLowestY = thisY;
319        else
320        end   
321     end
322     
323     if yCnt==yBlocks
324        %reset xStart, xStop positions and xCounter 
325         yCnt=0;
326         xCnt = xCnt+1; 
327     end
328           yCnt=yCnt+1;
329 end
330
331 %sprintf('Lowest (%d,%d), Highest (%d,%d)',currentLowestX, currentLowestY, currentHighestX,currentHighestY);
332
333 x = (currentLowestX*stepSize) - (stepSize-1);
334 y = (currentLowestY*stepSize) - (stepSize-1);
335 height = (currentHighestX*stepSize)-(stepSize-1)+(windowSize-1)-x;
336 width = (currentHighestY*stepSize)-(stepSize-1)+(windowSize-1)-y;
337 
338 %Crop out detected region based on colour detection and if possible, plot
339    if x>=0 && x<=xDim && y>=0 && y<=yDim
340       detectedRegion = imcrop(img,[y x width height]);
341       available=1;
342      % fprintf('Potential Traffic Sign Detected\n')
343    else
344        available=0;
345        detectedRegion = zeros(1,1);
346       % fprintf('No Traffic Sign Detected\n')
347    end
348    ROI = [y x width height]; %ROI co-ordinates to be sent back
349   %imshow(hsvImg);
350  
351end
