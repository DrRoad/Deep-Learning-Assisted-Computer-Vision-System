1function [r c] = detectCorner(img);
 2%detectCorner: Corner Detecter using Harris Method
 3%Input image should be grayscale
 4
 5%Validation
 6%if nargin ~= 1
 7%    error('detectCorner:Input_Argument_Error','This function works with 1 input argument -img- ')
 8%end
 9
10%Declare needed parameters
11dx = [-1 0 1; -1 0 1; -1 0 1]; % Sobel mask used for the derivative
12dy = dx'; %Transpose of dx
13sigma=2;
14supressionWindow = 20; %window size
15harrisHold = 1000000;
16
17[xDim, yDim, zDim] = size(img); %Check if image is coloured
18%{
19if (zDim>1)
20img = double(rgb2gray(img)); %Convert image to grayscale
21else
22%Do nothing
23end
24%}
25
26%Compute image FIRST ORDER DERIVATE of the image in both the x and y
27%direction by convolving it with the derivatice mask
28Ix = conv2(img, dx, 'same');    % Image derivatives
29Iy = conv2(img, dy, 'same'); 
30
31%Compute image SECOND ORDER DERIVATIVE by carrying out an element wise 
32%multiplication of the first order derivates
33Ixx = Ix.*Ix;
34Ixyp = Ix.*Iy;
35Iyy = Iy.*Iy;
36
37
38%To check the cornerness of each pixel of an image,a small window around 
39%that pixel is checked using a smooth circular window. To do this, we
40%filter all three second-order derivatives by convolving them with a 
41%Gaussian kernel.
42
43% Gaussian filter size = 6*sigma (+/- 3sigma). Minimum sixe 1x1.
44    g = fspecial('gaussian',max(1,fix(6*sigma)), sigma);
45    Ix2 = conv2(Ixx, g, 'same'); 
46    Iy2 = conv2(Iyy, g, 'same');
47    Ixy = conv2(Ixyp, g, 'same');
48
49%we compute the values of its trace and determinant. 
50    det = ((Ix2.*Iy2) - (Ixy.^2));
51    trace = (Ix2 + Iy2);
52%Find the harris response for each pixel
53% R = det(M) - k(trace M)^2
54k = 0.04; %Emperically computed and given by Harris
55R = det -(k*((trace).^2)); % Harris corner measure
56
57%Select corners
58%The Harris response for each pixel in the image is used to determine 
59%whether a pixel is a corner or not through non-maximal suppression
60%and thresholding
61
62    dilated = ordfilt2(R,supressionWindow^2,ones(supressionWindow)); % Grey-scale dilate.
63    %By thresholding the response array with the minimum desired Harris 
64    %response, which,we can calculate a binary response array for which 
65    %only pixels with a value greater than this threshold will be true
66    R = (R==dilated)&(R>harrisHold);       % Find maxima.
67    
68    [r,c] = find(R);                 % Find row,col coords.
69    %r = x
70    %c = y
71    
72    %figure, imagesc(img), axis image, colormap(gray), hold on
73    %plot(c,r,'go'), title('Detected Corners');
74    %fprintf('Corner Detection Carried Out\n')
75end %end of function
