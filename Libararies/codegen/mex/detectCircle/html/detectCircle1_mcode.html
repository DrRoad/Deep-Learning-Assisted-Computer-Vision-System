1function [circleAvailable strength] = detectCircle(img);
 2%detectCircle: Uses Circular Hough transform to detect the presence of a
 3%circle in a  image. Input image should be grayscale
 4
 5%{
 6STEPS
 71. Check to make sure recieved image is Black and White
 82. Detect
 93. Assign detection strength
10
11   Created on: 21st June, 2017
12   Author: Oluwole Oyetoke Jnr
13   Using MATLAB 2016
14%}
15
16%INPUT VALIDATION, INITIALIZATION AND IMAGE PREPARATION
17%--------------------------------------------------------------------------
18%if nargin ~= 1
19%    error('detectCircle:Input_Argument_Error','This function works with 1 input argument -img- ')
20%end
21
22[xDim, yDim, zDim] = size(img); %Check if image is coloured
23if (zDim>1)
24   img = rgb2gray(img);
25end
26
27lowerRadius = ceil((yDim*(0.1)));
28upperRadius =  ceil(yDim/2);
29%--------------------------------------------------------------------------
30
31%In order to optimize performance, divide search for circle into 2
32midPoint = ceil ((lowerRadius+upperRadius)/2);
33if midPoint>1
34[centers1,radii1 metric1] = imfindcircles(img,[lowerRadius midPoint],'ObjectPolarity','bright'); %Find all the circles with radius r pixels in the range [x1, x2].
35[centers2,radii2 metric2] = imfindcircles(img,[(midPoint+1) upperRadius],'ObjectPolarity','bright'); %Find all the circles with radius r pixels in the range [x3, x4].
36else
37    circleAvailable = 0; strength = 0;
38    %Not necessary, but code gen needs it
39    centers1 = zeros(1,1); radii1 = zeros(1,1); metric1 = zeros(1,1);
40    centers2 = zeros(1,1); radii2 = zeros(1,1); metric2 = zeros(1,1);
41end
42
43
44if isempty(radii1) && isempty(radii2) 
45    circleAvailable = 0; strength = 0;
46elseif isempty(radii1)==0 && isempty(radii2)==1 
47        centers = centers1; radii = radii1; metric=metric1;
48        strength = metric1(1,1); circleAvailable = 1;
49elseif isempty(radii1)==1 && isempty(radii2)==0
50       centers = centers2; radii = radii2; metric=metric2;
51       strength = metric2(1,1); circleAvailable = 1;
52else
53    %Find the strongest circle
54    if metric1(1) >= metric2(1)
55        centers = centers1; radii = radii1;metric=metric1;strength = metric1(1,1);    
56    else
57        centers = centers2;radii = radii2;metric=metric2;strength = metric2(1,1);
58    end
59    circleAvailable = 1;
60end %end of loop around possible circle area
61
62%    centresRadiiAndMetric = [centers radii metric]; %unused for now
63
64%--------------------------------------------------------------------------
65%SHAPE ANALYSIS - DRAW THE DETECTED CIRCLE
66%highlighted = drawDetectedCircle( img, xDim, yDim, centers, radii);
67%imshow(highlighted);
68%---------------------------------------------------------------------------
69
70
71%sprintf('Circle Detection Done. Availability: %d', circleAvailable)
72end %end of function
